    function setLang(lang) {
      document.getElementById('main-title').innerHTML = translations[lang].title;

      document.getElementById('buyer-title').textContent = translations[lang].buyer.title;

      document.getElementById('btn-en').classList.toggle('active', lang === 'en');

<script>
// SWITCH LANG
function switchLang() {
  var lang = document.documentElement.lang === "fr" ? "en" : "fr";
  document.documentElement.lang = lang;
  document.getElementById('langLabel').innerText = lang === "fr" ? "EN" : "FR";
  document.querySelectorAll('[data-fr],[data-en]').forEach(function(el) {
    var txt = el.getAttribute(lang === 'fr' ? 'data-fr' : 'data-en');
    if(txt) {
      if(el.placeholder !== undefined) el.placeholder = txt;
      else el.innerHTML = txt;
    }
  });
  document.querySelectorAll('select option').forEach(function(opt) {
    if(opt.getAttribute('data-fr') && opt.getAttribute('data-en')) {
      opt.textContent = lang === 'fr' ? opt.getAttribute('data-fr') : opt.getAttribute('data-en');
    }
  });
  document.querySelectorAll('.radio-group label').forEach(function(lbl) {
    var inpt = lbl.querySelector('input');
    if(inpt && inpt.getAttribute('data-fr') && inpt.getAttribute('data-en')) {
      lbl.lastChild.nodeValue = lang === 'fr' ? inpt.getAttribute('data-fr') : inpt.getAttribute('data-en');
    }
  });
  renderFaq(lang);
}
